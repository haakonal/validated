    def check_telemetry(telemetry: SatelliteTelemetry, max_battery_draw: float = 150.0) -> bool:
    """
    Ingests and checks a telemetry packet from the satellite.
    Raises ValidationError if any rule for the current mode is violated.
    """
    # Verify reaction wheel speeds are healthy across all modes
    check_reaction_wheels(telemetry.acs.reaction_wheel_speeds)

    if telemetry.mode == "charging":
        # Check if all solar panels are deployed
        panels_deployed = all(p.deployed for p in telemetry.panels)

        # Calculate total power generated by the panels
        total_gen = sum(p.power_generated for p in telemetry.panels)

        # Net power = generated - consumed (which is battery.current_draw when charging/consuming)
        net_power = total_gen - telemetry.battery.current_draw

        return validate_charging_telemetry(
            panels_deployed=panels_deployed,
            net_power=net_power,
            battery_status=telemetry.battery.status,
            battery_charge_level=telemetry.battery.charge_level,
            sun_pointing_deviation=telemetry.acs.pointing_deviation,
            reaction_wheel_speeds=telemetry.acs.reaction_wheel_speeds,
        )

    elif telemetry.mode == "data_collection":
        # Current draw must not exceed the max draw allowed by the battery
        power_margin = max_battery_draw - telemetry.battery.current_draw

        return validate_data_collection_telemetry(
            battery_temperature=telemetry.battery.temperature,
            ground_station_visible=telemetry.comms.ground_station_visible,
            power_margin=power_margin,
            battery_charge_level=telemetry.battery.charge_level,
            target_pointing_deviation=telemetry.acs.pointing_deviation,
            reaction_wheel_speeds=telemetry.acs.reaction_wheel_speeds,
        )

    # Safe or default mode: no additional constraints
    return True

    def validate_task(task: SlewTask | ImagingTask) -> bool:
    """
    Validates a task before it is committed to the spacecraft command queue.

    With Pydantic integration on the task models, validation happens automatically
    at construction time. This function exists as a unified API entry point and
    can be extended with cross-field or inter-model validation logic.
    """
    if isinstance(task, (SlewTask, ImagingTask)):
        return True
    raise ValueError(f"Unknown task type: {type(task)}")
